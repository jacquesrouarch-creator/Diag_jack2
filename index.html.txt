    // Pulse CAN
    const dot = document.getElementById('d-can');

        set('h-height', d.h); set('h-tilt', d.tilt.toFixed(1)); set('h-set', d.hset);
    }
    if(type === 'HYD') {

        set('s-arg', d.arg.toFixed(1)+'°'); set('s-ard', d.ard.toFixed(1)+'°');
        set('s-mode', d.smode); set('s-syn
